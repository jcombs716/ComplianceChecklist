    <script src="https://cdn.tailwindcss.com"></script>
    <script src="https://unpkg.com/lucide@latest"></script>
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    
    <!-- Firebase SDKs -->
    <script src="https://www.gstatic.com/firebasejs/11.6.1/firebase-app-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/11.6.1/firebase-auth-compat.js"></script>
    <script src="https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore-compat.js"></script>

    <style>
        @import url('https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@400;500;600;700;800&display=swap');
        body { font-family: 'Plus Jakarta Sans', sans-serif; }
        
        .glass { background: rgba(255, 255, 255, 0.7); backdrop-filter: blur(10px); border: 1px solid rgba(255, 255, 255, 0.3); }
        .custom-scrollbar::-webkit-scrollbar { width: 5px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #e2e8f0; border-radius: 10px; }
        
        .task-card { transition: transform 0.2s ease, box-shadow 0.2s ease; }
        .task-card:hover { transform: translateY(-2px); box-shadow: 0 10px 15px -3px rgba(0, 0, 0, 0.1); }
        
        @keyframes slideIn { from { transform: translateX(100%); opacity: 0; } to { transform: translateX(0); opacity: 1; } }
        .animate-slide-in { animation: slideIn 0.4s ease-out forwards; }
    </style>

<body class="bg-[#f8fafc] text-slate-900">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        // Configuration
        const firebaseConfig = JSON.parse(__firebase_config);
        const app = firebase.initializeApp(firebaseConfig);
        const auth = firebase.auth();
        const db = firebase.firestore();
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'ny-compliance-v3';

        const Icon = ({ name, size = 18, className = "" }) => <i data-lucide={name} style={{ width: size, height: size }} className={className}></i>;

        // Master template that you (Admin) would supply
        const INITIAL_MASTER_TEMPLATES = [
            { category: "Facilities", subcategory: "Structural", name: "Visual Inspection", frequency: "Annually", authority: "NYSED", action: "Inspection", performedBy: "Team", citation: "8 NYCRR ยง 155.4" },
            { category: "Fire Safety", subcategory: "General", name: "Annual Fire Safety Inspection", frequency: "Annually", authority: "NYSED", action: "Inspection", performedBy: "Code Official", citation: "Ed. Law ยง 807-A" },
            { category: "Environmental", subcategory: "Water", name: "Lead in Water Testing", frequency: "Every 3 Years", authority: "NYS DOH", action: "Sampling", performedBy: "Lab", citation: "10 NYCRR ยง 67-4" }
        ];

        function App() {
            const [user, setUser] = useState(null);
            const [loading, setLoading] = useState(true);
            const [isFirstRun, setIsFirstRun] = useState(false);
            const [onboardingStep, setOnboardingStep] = useState(1);
            const [showHelp, setShowHelp] = useState(false);
            
            // District State
            const [districtData, setDistrictData] = useState({ name: '', adminName: '', adminTitle: '', buildings: [] });
            const [userTasks, setUserTasks] = useState([]);
            const [globalTemplates, setGlobalTemplates] = useState([]);
            
            // UI State
            const [selectedBuilding, setSelectedBuilding] = useState('');
            const [searchTerm, setSearchTerm] = useState('');
            const [expandedTaskId, setExpandedTaskId] = useState(null);
            const [showGlobalPrompt, setShowGlobalPrompt] = useState(false);

            useEffect(() => {
                const initAuth = async () => {
                    if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                        await auth.signInWithCustomToken(__initial_auth_token);
                    } else {
                        await auth.signInAnonymously();
                    }
                };
                initAuth();
                const unsub = auth.onAuthStateChanged(u => { setUser(u); if(!u) setLoading(false); });
                return unsub;
            }, []);

            useEffect(() => {
                if (!user) return;

                const userConfigRef = db.collection('artifacts').doc(appId).collection('users').doc(user.uid).collection('settings').doc('config');
                const userTasksRef = db.collection('artifacts').doc(appId).collection('users').doc(user.uid).collection('tasks');
                const globalRef = db.collection('artifacts').doc(appId).collection('public').doc('data').collection('templates');

                const unsubConfig = userConfigRef.onSnapshot(doc => {
                    if (doc.exists) {
                        const data = doc.data();
                        setDistrictData(data);
                        if (!selectedBuilding && data.buildings?.length > 0) setSelectedBuilding(data.buildings[0]);
                        setIsFirstRun(false);
                    } else {
                        setIsFirstRun(true);
                    }
                    setLoading(false);
                }, () => setLoading(false));

                const unsubUserTasks = userTasksRef.onSnapshot(snap => {
                    setUserTasks(snap.docs.map(d => ({...d.data(), id: d.id})));
                    setTimeout(() => window.lucide?.createIcons(), 100);
                });

                const unsubGlobal = globalRef.onSnapshot(snap => {
                    const templates = snap.docs.map(d => ({...d.data(), id: d.id}));
                    setGlobalTemplates(templates);
                });

                return () => { unsubConfig(); unsubUserTasks(); unsubGlobal(); };
            }, [user]);

            const handleFinishOnboarding = async () => {
                if (!user) return;
                setLoading(true);
                const userConfigRef = db.collection('artifacts').doc(appId).collection('users').doc(user.uid).collection('settings').doc('config');
                const userTasksRef = db.collection('artifacts').doc(appId).collection('users').doc(user.uid).collection('tasks');

                try {
                    await userConfigRef.set({ ...districtData, setupComplete: true });
                    const batch = db.batch();
                    districtData.buildings.forEach(b => {
                        INITIAL_MASTER_TEMPLATES.forEach((m, idx) => {
                            const tid = `${b}-${idx}`.replace(/\s+/g, '-').toLowerCase();
                            batch.set(userTasksRef.doc(tid), {
                                ...m,
                                building: b,
                                completed: false,
                                lastCompleted: null
                            });
                        });
                    });
                    await batch.commit();
                } catch (e) { console.error(e); }
                setLoading(false);
            };

            const toggleTask = async (task) => {
                const ref = db.collection('artifacts').doc(appId).collection('users').doc(user.uid).collection('tasks').doc(task.id);
                await ref.update({
                    completed: !task.completed,
                    lastCompleted: !task.completed ? new Date().toLocaleDateString() : null
                });
            };

            const filteredTasks = useMemo(() => {
                return userTasks.filter(t => 
                    t.building === selectedBuilding && 
                    (t.name.toLowerCase().includes(searchTerm.toLowerCase()) || 
                     t.category.toLowerCase().includes(searchTerm.toLowerCase()))
                );
            }, [userTasks, selectedBuilding, searchTerm]);

            if (loading) return (
                <div className="h-screen flex flex-col items-center justify-center bg-white">
                    <div className="w-12 h-12 border-4 border-indigo-600 border-t-transparent rounded-full animate-spin mb-4"></div>
                    <p className="font-bold text-slate-400 uppercase tracking-widest text-xs">Syncing District Data...</p>
                </div>
            );

            if (isFirstRun) {
                return (
                    <div className="min-h-screen bg-[#6366f1] flex items-center justify-center p-4">
                        <div className="max-w-xl w-full glass rounded-[2.5rem] p-10 shadow-2xl animate-in zoom-in duration-500">
                            <div className="flex items-center gap-3 mb-10">
                                <div className="w-12 h-12 bg-indigo-600 rounded-2xl flex items-center justify-center text-white shadow-lg shadow-indigo-200">
                                    <Icon name="shield-check" size={24} />
                                </div>
                                <div>
                                    <h1 className="text-xl font-extrabold tracking-tight text-slate-800">Compliance Setup</h1>
                                    <p className="text-xs text-slate-500 font-bold uppercase tracking-wider">Step {onboardingStep} of 3</p>
                                </div>
                            </div>

                            {onboardingStep === 1 && (
                                <div className="space-y-6">
                                    <h2 className="text-3xl font-black text-slate-800 leading-tight">Identify your <br/><span className="text-indigo-600">School District.</span></h2>
                                    <div className="space-y-4">
                                        <div className="group">
                                            <label className="block text-[10px] font-black text-slate-400 uppercase mb-2 ml-1">Official District Name</label>
                                            <input type="text" className="w-full bg-white/50 border-2 border-slate-100 rounded-2xl p-4 focus:border-indigo-500 outline-none transition-all font-medium" placeholder="e.g. Hudson Valley Central" value={districtData.name} onChange={e => setDistrictData({...districtData, name: e.target.value})} />
                                        </div>
                                        <div className="grid grid-cols-2 gap-4">
                                            <div>
                                                <label className="block text-[10px] font-black text-slate-400 uppercase mb-2 ml-1">Full Name</label>
                                                <input type="text" className="w-full bg-white/50 border-2 border-slate-100 rounded-2xl p-4 focus:border-indigo-500 outline-none transition-all text-sm font-medium" placeholder="Jane Smith" value={districtData.adminName} onChange={e => setDistrictData({...districtData, adminName: e.target.value})} />
                                            </div>
                                            <div>
                                                <label className="block text-[10px] font-black text-slate-400 uppercase mb-2 ml-1">Job Title</label>
                                                <input type="text" className="w-full bg-white/50 border-2 border-slate-100 rounded-2xl p-4 focus:border-indigo-500 outline-none transition-all text-sm font-medium" placeholder="Dir. of Facilities" value={districtData.adminTitle} onChange={e => setDistrictData({...districtData, adminTitle: e.target.value})} />
                                            </div>
                                        </div>
                                    </div>
                                    <button onClick={() => setOnboardingStep(2)} disabled={!districtData.name} className="w-full bg-indigo-600 text-white font-extrabold py-5 rounded-3xl shadow-xl shadow-indigo-200 hover:scale-[1.02] active:scale-95 transition-all disabled:opacity-50 mt-4">Next: Map Buildings</button>
                                </div>
                            )}

                            {onboardingStep === 2 && (
                                <div className="space-y-6">
                                    <h2 className="text-3xl font-black text-slate-800 leading-tight">List your <br/><span className="text-indigo-600">Buildings.</span></h2>
                                    <div className="space-y-3 max-h-48 overflow-y-auto pr-2 custom-scrollbar">
                                        {districtData.buildings.map((b, i) => (
                                            <div key={i} className="flex items-center justify-between p-4 bg-white rounded-2xl border border-slate-100 shadow-sm animate-in slide-in-from-bottom-2">
                                                <span className="font-bold text-slate-700">{b}</span>
                                                <button onClick={() => setDistrictData({...districtData, buildings: districtData.buildings.filter((_, idx) => idx !== i)})} className="text-slate-300 hover:text-red-500 transition-colors">
                                                    <Icon name="trash-2" size={16}/>
                                                </button>
                                            </div>
                                        ))}
                                    </div>
                                    <div className="flex gap-2">
                                        <input id="bIn" type="text" placeholder="Add a building..." className="flex-1 bg-white border-2 border-slate-100 rounded-2xl p-4 outline-none focus:border-indigo-500 font-medium" onKeyDown={e => {if(e.key==='Enter' && e.target.value){ setDistrictData({...districtData, buildings: [...districtData.buildings, e.target.value]}); e.target.value=''; }}} />
                                        <button onClick={() => {const el=document.getElementById('bIn'); if(el.value){setDistrictData({...districtData, buildings:[...districtData.buildings, el.value]}); el.value='';}}} className="bg-indigo-50 text-indigo-600 w-14 rounded-2xl flex items-center justify-center hover:bg-indigo-100"><Icon name="plus" /></button>
                                    </div>
                                    <div className="flex gap-3 pt-4">
                                        <button onClick={() => setOnboardingStep(1)} className="flex-1 bg-slate-100 text-slate-500 font-bold py-5 rounded-3xl">Back</button>
                                        <button onClick={() => setOnboardingStep(3)} disabled={districtData.buildings.length===0} className="flex-2 bg-indigo-600 text-white font-extrabold py-5 px-10 rounded-3xl shadow-xl shadow-indigo-200 disabled:opacity-50">Generate Portal</button>
                                    </div>
                                </div>
                            )}

                            {onboardingStep === 3 && (
                                <div className="text-center space-y-6 py-4">
                                    <div className="w-24 h-24 bg-green-100 text-green-600 rounded-full flex items-center justify-center mx-auto mb-6 animate-bounce">
                                        <Icon name="check-circle" size={48} />
                                    </div>
                                    <h2 className="text-4xl font-black text-slate-800">You're all set!</h2>
                                    <p className="text-slate-500 font-medium">We've customized your compliance matrix for <strong>{districtData.buildings.length} buildings</strong>. Your dashboard is ready to secure your district.</p>
                                    <button onClick={handleFinishOnboarding} className="w-full bg-indigo-600 text-white font-extrabold py-6 rounded-[2rem] shadow-2xl shadow-indigo-300 hover:scale-[1.03] transition-transform">Enter My Dashboard</button>
                                </div>
                            )}
                        </div>
                    </div>
                );
            }

            return (
                <div className="flex flex-col lg:flex-row h-screen overflow-hidden">
                    {/* Sidebar */}
                    <aside className="w-full lg:w-80 bg-slate-900 flex flex-col border-r border-slate-800">
                        <div className="p-8">
                            <div className="flex items-center gap-4 text-white mb-10">
                                <div className="w-12 h-12 bg-gradient-to-br from-indigo-500 to-indigo-700 rounded-2xl flex items-center justify-center font-black text-xl shadow-lg shadow-indigo-500/20">
                                    {districtData.name?.charAt(0) || "D"}
                                </div>
                                <div className="overflow-hidden">
                                    <h1 className="font-extrabold text-sm uppercase tracking-tight truncate">{districtData.name}</h1>
                                    <div className="flex items-center gap-2">
                                        <div className="w-1.5 h-1.5 rounded-full bg-green-500 animate-pulse"></div>
                                        <span className="text-[10px] text-slate-400 font-bold uppercase tracking-widest">Live Cloud Sync</span>
                                    </div>
                                </div>
                            </div>

                            <nav className="space-y-6">
                                <div>
                                    <label className="text-[10px] font-black text-slate-500 uppercase tracking-widest block px-4 mb-3">Buildings</label>
                                    <div className="space-y-1">
                                        {districtData.buildings.map(b => (
                                            <button key={b} onClick={() => setSelectedBuilding(b)} className={`w-full text-left px-4 py-3 rounded-xl text-sm transition-all flex items-center justify-between group ${selectedBuilding === b ? 'bg-indigo-600 text-white font-bold shadow-indigo-500/30 shadow-lg translate-x-1' : 'text-slate-400 hover:bg-slate-800/50 hover:text-slate-200'}`}>
                                                <span className="truncate">{b}</span>
                                                <Icon name="chevron-right" size={14} className={selectedBuilding === b ? 'opacity-100' : 'opacity-0 group-hover:opacity-100'} />
                                            </button>
                                        ))}
                                    </div>
                                </div>
                            </nav>
                        </div>

                        <div className="mt-auto p-6 border-t border-slate-800 bg-slate-950/40">
                            <div className="flex flex-col gap-4">
                                <button onClick={() => setShowHelp(true)} className="flex items-center gap-3 text-slate-400 hover:text-white transition-colors text-xs font-bold uppercase tracking-widest px-2">
                                    <Icon name="help-circle" size={16} />
                                    Deployment Help
                                </button>
                                <div className="flex items-center gap-4">
                                    <div className="w-10 h-10 rounded-2xl bg-slate-800 border border-slate-700 flex items-center justify-center text-indigo-400">
                                        <Icon name="user" />
                                    </div>
                                    <div className="min-w-0">
                                        <p className="text-sm font-bold text-white truncate">{districtData.adminName}</p>
                                        <p className="text-[10px] text-slate-500 font-bold uppercase truncate">{districtData.adminTitle}</p>
                                    </div>
                                </div>
                            </div>
                        </div>
                    </aside>

                    {/* Main Content */}
                    <main className="flex-1 flex flex-col bg-[#f8fafc] overflow-hidden">
                        {globalTemplates.length > 0 && (
                            <div className="bg-indigo-600 text-white px-6 py-3 flex items-center justify-between animate-in slide-in-from-top duration-500">
                                <div className="flex items-center gap-3">
                                    <Icon name="megaphone" />
                                    <span className="text-sm font-bold">New NY Compliance Requirements available for review.</span>
                                </div>
                                <button onClick={() => setShowGlobalPrompt(true)} className="bg-white text-indigo-600 text-xs font-black uppercase px-4 py-1.5 rounded-full hover:bg-indigo-50 transition-colors">Review Changes</button>
                            </div>
                        )}

                        <header className="p-8 pb-4 flex flex-col md:flex-row items-center justify-between gap-6">
                            <div>
                                <h2 className="text-3xl font-black text-slate-800">{selectedBuilding}</h2>
                                <p className="text-slate-400 text-sm font-medium">Compliance Overview & Task Tracking</p>
                            </div>
                            <div className="relative w-full md:w-80">
                                <input type="text" placeholder="Filter requirements..." className="w-full bg-white border border-slate-200 rounded-2xl py-3.5 pl-12 pr-4 outline-none focus:ring-4 focus:ring-indigo-500/5 focus:border-indigo-500 transition-all text-sm font-medium shadow-sm" value={searchTerm} onChange={e => setSearchTerm(e.target.value)} />
                                <div className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-300">
                                    <Icon name="search" />
                                </div>
                            </div>
                        </header>

                        <div className="flex-1 overflow-y-auto p-8 pt-4 custom-scrollbar">
                            {filteredTasks.length === 0 ? (
                                <div className="h-64 border-2 border-dashed border-slate-200 rounded-[2.5rem] flex flex-col items-center justify-center text-slate-400">
                                    <Icon name="clipboard-check" size={48} className="mb-4 opacity-20" />
                                    <p className="font-bold uppercase tracking-widest text-xs">No tasks found in this view</p>
                                </div>
                            ) : (
                                <div className="grid grid-cols-1 xl:grid-cols-2 gap-4">
                                    {filteredTasks.map(task => (
                                        <div key={task.id} className={`task-card bg-white rounded-3xl border p-6 flex flex-col gap-4 ${task.completed ? 'border-green-100' : 'border-slate-100 shadow-sm shadow-slate-200/50'}`}>
                                            <div className="flex items-start justify-between gap-4">
                                                <div className="flex items-center gap-4">
                                                    <button onClick={() => toggleTask(task)} className={`w-12 h-12 rounded-2xl border-2 flex items-center justify-center transition-all ${task.completed ? 'bg-green-500 border-green-500 text-white shadow-lg shadow-green-100' : 'bg-white border-slate-100 hover:border-indigo-200'}`}>
                                                        <Icon name={task.completed ? "check" : "circle"} size={20} />
                                                    </button>
                                                    <div>
                                                        <h4 className={`font-extrabold text-slate-800 ${task.completed ? 'line-through opacity-40' : ''}`}>{task.name}</h4>
                                                        <div className="flex flex-wrap items-center gap-2 mt-1">
                                                            <span className="text-[10px] font-black uppercase tracking-widest bg-slate-100 text-slate-500 px-2 py-0.5 rounded-md">{task.category}</span>
                                                            <span className="text-[10px] font-black uppercase tracking-widest bg-indigo-50 text-indigo-500 px-2 py-0.5 rounded-md">{task.frequency}</span>
                                                        </div>
                                                    </div>
                                                </div>
                                                <button onClick={() => setExpandedTaskId(expandedTaskId === task.id ? null : task.id)} className="w-8 h-8 flex items-center justify-center text-slate-300 hover:text-slate-600 transition-colors">
                                                    <Icon name={expandedTaskId === task.id ? "chevron-up" : "more-horizontal"} />
                                                </button>
                                            </div>

                                            {expandedTaskId === task.id && (
                                                <div className="pt-4 border-t border-slate-50 space-y-4 animate-in fade-in duration-300">
                                                    <div className="grid grid-cols-2 gap-4 text-xs">
                                                        <div className="space-y-1">
                                                            <p className="text-slate-400 font-bold uppercase tracking-tighter text-[9px]">Authority</p>
                                                            <p className="font-bold text-slate-700">{task.authority}</p>
                                                        </div>
                                                        <div className="space-y-1">
                                                            <p className="text-slate-400 font-bold uppercase tracking-tighter text-[9px]">Citation</p>
                                                            <p className="font-bold text-slate-700">{task.citation}</p>
                                                        </div>
                                                    </div>
                                                    <div className="bg-slate-50 rounded-2xl p-4 text-[13px] font-medium text-slate-600 leading-relaxed border border-slate-100">
                                                        <span className="text-[9px] font-black uppercase text-indigo-400 block mb-1">Required Action</span>
                                                        {task.action}
                                                    </div>
                                                    {task.lastCompleted && (
                                                        <div className="flex items-center gap-2 text-[10px] font-bold text-green-600 bg-green-50 w-fit px-3 py-1.5 rounded-full">
                                                            <Icon name="calendar" size={12} />
                                                            Last Performed: {task.lastCompleted}
                                                        </div>
                                                    )}
                                                </div>
                                            )}
                                        </div>
                                    ))}
                                </div>
                            )}
                        </div>
                    </main>

                    {/* Deployment Help Modal */}
                    {showHelp && (
                        <div className="fixed inset-0 z-[60] flex items-center justify-center p-6">
                            <div className="absolute inset-0 bg-slate-900/60 backdrop-blur-md" onClick={() => setShowHelp(false)}></div>
                            <div className="relative glass w-full max-w-2xl rounded-[3rem] p-10 shadow-2xl animate-in zoom-in duration-300 max-h-[90vh] overflow-y-auto custom-scrollbar text-slate-800">
                                <h2 className="text-3xl font-black mb-6">How to Launch This App</h2>
                                
                                <div className="space-y-8">
                                    <section>
                                        <h3 className="text-indigo-600 font-black uppercase tracking-widest text-xs mb-3">Option 1: Vercel (Recommended)</h3>
                                        <ol className="list-decimal list-inside space-y-2 text-sm font-medium text-slate-600">
                                            <li>Save this code as a file named <code className="bg-slate-100 px-1 rounded">index.html</code>.</li>
                                            <li>Go to <a href="https://vercel.com/deploy" target="_blank" className="text-indigo-500 underline">vercel.com/deploy</a>.</li>
                                            <li>Instead of "Import", look for the <strong className="text-slate-800">"Drag and Drop"</strong> area at the bottom or middle of the screen.</li>
                                            <li>Simply drag your <code className="bg-slate-100 px-1 rounded">index.html</code> file onto that area.</li>
                                            <li>It will instantly deploy and give you a public URL.</li>
                                        </ol>
                                    </section>

                                    <section>
                                        <h3 className="text-indigo-600 font-black uppercase tracking-widest text-xs mb-3">Option 2: Netlify Drop</h3>
                                        <ol className="list-decimal list-inside space-y-2 text-sm font-medium text-slate-600">
                                            <li>Go to <a href="https://app.netlify.com/drop" target="_blank" className="text-indigo-500 underline">netlify.com/drop</a>.</li>
                                            <li>Drag your folder containing <code className="bg-slate-100 px-1 rounded">index.html</code> onto the page.</li>
                                            <li>Your site is live immediately.</li>
                                        </ol>
                                    </section>

                                    <section className="bg-amber-50 p-6 rounded-3xl border border-amber-100">
                                        <h3 className="text-amber-700 font-black uppercase tracking-widest text-xs mb-2">Important Security Note</h3>
                                        <p className="text-xs text-amber-600 font-medium">Since this runs in a browser, your Firebase configuration is public. Ensure your <strong className="text-amber-800">Firestore Security Rules</strong> are set to only allow users to read/write to their own <code className="bg-amber-100 px-1 rounded">/users/{`{userId}`}</code> paths.</p>
                                    </section>
                                </div>

                                <button onClick={() => setShowHelp(false)} className="w-full mt-10 py-5 bg-slate-900 text-white rounded-[2rem] font-bold">Got it, thanks!</button>
                            </div>
                        </div>
                    )}

                    {/* Global Requirement Modal */}
                    {showGlobalPrompt && (
                        <div className="fixed inset-0 z-50 flex items-center justify-center p-6">
